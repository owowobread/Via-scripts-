// ==UserScript==
// @name         Instagram Media Viewer - Ultimate Nuke v12 (Master Toggle)
// @namespace    https://viayoo.com/
// @version      1.8
// @description  Total nav eradication, 100dvh true fullscreen, Smart Auto-Fit, Quick Save, Speed Control, Normal Mode Toggle
// @author       You
// @run-at       document-idle
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function() {
    // --- GLOBAL STATE VARIABLES ---
    let currentPlaybackSpeed = 1;
    let isMasterEnabled = true;
    let isNuked = false;

    // --- 1. GLOBAL NAV ERADICATION (ALWAYS ACTIVE UNLESS TOGGLED) ---
    const globalStyle = document.createElement('style');
    globalStyle.id = 'ig-global-override';
    globalStyle.innerHTML = `
        /* TARGET SPECIFIC LINKS INSIDE THE NAV BAR TO HIDE THE ENTIRE ROW */
        div:has(> a[href^="/explore"]),
        div:has(> a[href^="/reels"]),
        div:has(> a[href="/explore/"]),
        div:has(> div > a[href="/explore/"]),
        [role="tablist"],
        [role="navigation"] > div > div > div:has(a) { 
            display: none !important; 
            opacity: 0 !important;
            pointer-events: none !important;
            height: 0 !important;
            min-height: 0 !important;
            margin: 0 !important;
            padding: 0 !important;
        }
        
        /* REMOVE PADDING & FORCE FULL HEIGHT */
        html, body, #react-root, main, section {
            padding-bottom: 0 !important;
            margin-bottom: 0 !important;
            background-color: #000 !important;
        }
    `;
    document.head.appendChild(globalStyle);

    // --- 2. JAVASCRIPT BACKGROUND ENGINE ---
    setInterval(() => {
        if (!isMasterEnabled) return; // Halt all background enforcement if Normal Mode is ON

        const navLinks = document.querySelectorAll('a[href="/explore/"], a[href^="/reels/"]');
        navLinks.forEach(link => {
            let curr = link;
            while(curr && curr !== document.body) {
                const style = window.getComputedStyle(curr);
                if (style.position === 'fixed' || style.position === 'sticky' || curr.getAttribute('role') === 'tablist') {
                    // Tag it so we can revert it later
                    curr.setAttribute('data-ig-script-hidden', 'true');
                    curr.style.setProperty('display', 'none', 'important');
                    curr.style.setProperty('height', '0', 'important');
                    curr.style.setProperty('pointer-events', 'none', 'important');
                    break;
                }
                curr = curr.parentElement;
            }
        });

        const mediaElements = document.querySelectorAll('article video, article img:not([alt*="profile" i]), [role="presentation"] video, [role="presentation"] img:not([alt*="profile" i])');
        mediaElements.forEach(media => {
            let ratio = null;
            if (media.tagName === 'VIDEO') {
                if (media.videoWidth && media.videoHeight) {
                    ratio = media.videoWidth / media.videoHeight;
                }
                if (media.playbackRate !== currentPlaybackSpeed) {
                    media.playbackRate = currentPlaybackSpeed;
                }
            } else if (media.tagName === 'IMG' && media.naturalWidth && media.naturalHeight) {
                ratio = media.naturalWidth / media.naturalHeight;
            }

            if (ratio !== null) {
                media.setAttribute('data-smart-fit', ratio <= 0.65 ? 'cover' : 'contain');
            }
        });
    }, 500);

    // --- 3. CREATE CONTROL PANEL ---
    const panel = document.createElement('div');
    panel.id = 'ig-control-panel';
    panel.style.cssText = 'position:fixed;bottom:24px;right:24px;z-index:999999;display:flex;flex-direction:column;gap:12px;';

    const btnBase = 'width:36px;height:36px;cursor:pointer;display:flex;align-items:center;justify-content:center;background:transparent;border:none;transition:all 0.2s ease;opacity:0.4;padding:0;outline:none;';
    const btnHover = 'opacity:1;';

    // --- TOAST NOTIFICATION SYSTEM ---
    const toast = document.createElement('div');
    toast.style.cssText = 'position:fixed;top:50%;left:50%;transform:translate(-50%, -50%);background:rgba(0,0,0,0.6);backdrop-filter:blur(10px);-webkit-backdrop-filter:blur(10px);border:1px solid rgba(255,255,255,0.1);color:white;padding:12px 24px;border-radius:30px;font-family:sans-serif;font-weight:bold;font-size:14px;z-index:1000000;opacity:0;pointer-events:none;transition:opacity 0.3s;';
    document.body.appendChild(toast);

    let toastTimeout;
    const showToast = (msg) => {
        toast.innerText = msg;
        toast.style.opacity = '1';
        clearTimeout(toastTimeout);
        toastTimeout = setTimeout(() => toast.style.opacity = '0', 1200);
    };

    // --- A. SAVE TO PROFILE BUTTON ---
    const saveBtn = document.createElement('button');
    saveBtn.style.cssText = btnBase;
    saveBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="rgba(255,255,255,0.9)" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M19 21l-7-5-7 5V5a2 2 0 0 1 2-2h10a2 2 0 0 1 2 2z"></path></svg>`;
    saveBtn.onmouseenter = () => saveBtn.style.cssText = btnBase + btnHover;
    saveBtn.onmouseleave = () => saveBtn.style.cssText = btnBase;

    saveBtn.onclick = () => {
        const mediaElements = Array.from(document.querySelectorAll('video, img:not([alt*="profile" i])'));
        let activeMedia = null;
        let minDistance = Infinity;
        const centerY = window.innerHeight / 2;

        mediaElements.forEach(media => {
            const rect = media.getBoundingClientRect();
            if (rect.height === 0 || rect.width === 0) return; 
            const mediaCenter = rect.top + rect.height / 2;
            const distance = Math.abs(centerY - mediaCenter);
            if (distance < minDistance) {
                minDistance = distance;
                activeMedia = media;
            }
        });

        if (activeMedia) {
            let curr = activeMedia;
            let saveIcon = null;
            
            while (curr && curr !== document.body) {
                saveIcon = curr.querySelector('svg[aria-label="Save"], svg[aria-label="Remove"]');
                if (!saveIcon) {
                    const svgs = curr.querySelectorAll('svg');
                    for (let svg of svgs) {
                        const poly = svg.querySelector('polygon');
                        if (poly && poly.getAttribute('points') && poly.getAttribute('points').includes('20 21 12')) {
                            saveIcon = svg;
                            break;
                        }
                    }
                }
                if (saveIcon) break;
                curr = curr.parentElement;
            }

            if (saveIcon) {
                const btn = saveIcon.closest('button') || saveIcon.closest('[role="button"]') || saveIcon.closest('a') || saveIcon.parentNode;
                if (btn && typeof btn.click === 'function') {
                    btn.click();
                    const actionLabel = saveIcon.getAttribute('aria-label');
                    if (actionLabel === 'Save') showToast('Saved to Profile');
                    else if (actionLabel === 'Remove') showToast('Removed from Profile');
                    else showToast('Bookmark Toggled');
                    saveBtn.style.transform = 'scale(0.8)';
                    setTimeout(() => saveBtn.style.transform = 'scale(1)', 150);
                    return;
                }
            }
        }
        showToast('Save button not found on this post');
    };
    panel.appendChild(saveBtn);

    // --- B. NUKE BUTTON ---
    const hideBtn = document.createElement('button');
    hideBtn.style.cssText = btnBase;
    hideBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="rgba(255,255,255,0.9)" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M9.88 9.88a3 3 0 1 0 4.24 4.24"/><path d="M10.73 5.08A10.43 10.43 0 0 1 12 5c7 0 10 7 10 7a13.16 13.16 0 0 1-1.67 2.68"/><path d="M6.61 6.61A13.526 13.526 0 0 0 2 12s3 7 10 7a9.74 9.74 0 0 0 5.39-1.61"/><line x1="2" x2="22" y1="2" y2="22"/></svg>`;
    hideBtn.onmouseenter = () => hideBtn.style.cssText = btnBase + btnHover;
    hideBtn.onmouseleave = () => hideBtn.style.cssText = btnBase + (isNuked ? 'opacity:0.8;' : '');

    hideBtn.onclick = () => {
        isNuked = !isNuked;
        let style = document.getElementById('ig-nuke-ui');
        
        if (!style) {
            style = document.createElement('style');
            style.id = 'ig-nuke-ui';
            style.innerHTML = `
                html, body, #react-root, main, section { overflow-x: hidden !important; scroll-snap-type: y mandatory !important; }
                canvas { display: none !important; }
                span, p, h1, h2, h3, h4, h5, h6, figcaption, time, label, [dir="auto"] { display: none !important; color: transparent !important; }
                header, nav, footer, aside, [role="navigation"], [role="banner"], [role="dialog"], [role="region"] { display: none !important; }
                svg:not(#ig-control-panel svg), input, textarea, form, [role="menu"] { display: none !important; }
                button:not(:has(video)):not(:has(img)):not(#ig-control-panel button), [role="button"]:not(:has(video)):not(:has(img)) { display: none !important; }
                a:not(:has(video)):not(:has(img)) { display: none !important; }
                img[alt*="profile" i], img[alt*="icon" i] { display: none !important; }
                
                article div:not(:has(video)):not(:has(img:not([alt*="profile" i]))),
                [role="presentation"] div:not(:has(video)):not(:has(img:not([alt*="profile" i]))) { background: transparent !important; }

                article {
                    width: 100vw !important; height: 100dvh !important; margin: 0 !important; padding: 0 !important; border: none !important;
                    scroll-snap-align: start !important; scroll-snap-stop: always !important; display: flex !important;
                    align-items: center !important; justify-content: center !important; position: relative !important; background: #000 !important;
                }
                article video, article img:not([alt*="profile" i]), [role="presentation"] video, [role="presentation"] img:not([alt*="profile" i]) {
                    position: absolute !important; top: 50% !important; left: 50% !important;
                    transform: translate(-50%, -50%) !important; z-index: 99998 !important; border-radius: 0 !important;
                }
            `;
            document.head.appendChild(style);
        }
        
        style.disabled = !isNuked;
        hideBtn.style.transform = 'scale(0.9)';
        setTimeout(() => hideBtn.style.transform = 'scale(1)', 150);
        showToast(isNuked ? 'UI Nuked' : 'UI Restored');
    };
    panel.appendChild(hideBtn);

    // --- C. ASPECT RATIO BUTTON ---
    const ratioBtn = document.createElement('button');
    ratioBtn.style.cssText = btnBase;
    ratioBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="rgba(255,255,255,0.9)" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="15 3 21 3 21 9"/><polyline points="9 21 3 21 3 15"/><line x1="21" x2="14" y1="3" y2="10"/><line x1="3" x2="10" y1="21" y2="14"/></svg>`;
    ratioBtn.onmouseenter = () => ratioBtn.style.cssText = btnBase + btnHover;
    ratioBtn.onmouseleave = () => ratioBtn.style.cssText = btnBase;

    const ratioStyle = document.createElement('style');
    ratioStyle.id = 'ig-ratio-override';
    document.head.appendChild(ratioStyle);

    const modes = [
        { name: "Mode: Auto (Smart Fit)", css: `article video[data-smart-fit="cover"], article img[data-smart-fit="cover"], [role="presentation"] video[data-smart-fit="cover"], [role="presentation"] img[data-smart-fit="cover"] { width: 100dvw !important; height: 100dvh !important; object-fit: cover !important; } article video[data-smart-fit="contain"], article img[data-smart-fit="contain"], [role="presentation"] video[data-smart-fit="contain"], [role="presentation"] img[data-smart-fit="contain"], article video:not([data-smart-fit]), article img:not([data-smart-fit]) { width: 100vw !important; height: auto !important; max-height: 100dvh !important; object-fit: contain !important; }` },
        { name: "Mode: Force Fill", css: `article video, article img:not([alt*="profile" i]), [role="presentation"] video, [role="presentation"] img:not([alt*="profile" i]) { width: 100dvw !important; height: 100dvh !important; object-fit: cover !important; }` },
        { name: "Mode: Force Original", css: `article video, article img:not([alt*="profile" i]), [role="presentation"] video, [role="presentation"] img:not([alt*="profile" i]) { width: 100vw !important; height: auto !important; max-height: 100dvh !important; object-fit: contain !important; }` }
    ];

    let currentMode = 0; 
    ratioStyle.innerHTML = modes[currentMode].css; 

    ratioBtn.onclick = () => {
        currentMode = (currentMode + 1) % modes.length; 
        ratioStyle.innerHTML = modes[currentMode].css;
        ratioBtn.style.transform = 'scale(0.9)';
        setTimeout(() => ratioBtn.style.transform = 'scale(1)', 150);
        showToast(modes[currentMode].name);
    };
    panel.appendChild(ratioBtn);

    // --- D. PLAYBACK SPEED BUTTON ---
    const speedBtnTextStyling = 'font-family: sans-serif; font-size: 14px; font-weight: bold; color: rgba(255,255,255,0.9); text-shadow: 0px 1px 3px rgba(0,0,0,0.8);';
    const speedBtn = document.createElement('button');
    speedBtn.style.cssText = btnBase + speedBtnTextStyling;
    speedBtn.innerText = '1x'; 

    speedBtn.onmouseenter = () => speedBtn.style.cssText = btnBase + speedBtnTextStyling + btnHover;
    speedBtn.onmouseleave = () => speedBtn.style.cssText = btnBase + speedBtnTextStyling;

    const speeds = [0.25, 0.5, 1, 2, 3];
    let speedIdx = 2; // Default to 1x

    speedBtn.onclick = () => {
        speedIdx = (speedIdx + 1) % speeds.length;
        currentPlaybackSpeed = speeds[speedIdx];
        speedBtn.innerText = currentPlaybackSpeed + 'x';
        
        document.querySelectorAll('video').forEach(video => {
            video.playbackRate = currentPlaybackSpeed;
        });

        speedBtn.style.transform = 'scale(0.9)';
        setTimeout(() => speedBtn.style.transform = 'scale(1)', 150);
        showToast(`Speed: ${currentPlaybackSpeed}x`);
    };
    panel.appendChild(speedBtn);

    // --- E. MASTER NORMAL MODE TOGGLE BUTTON ---
    const masterBtn = document.createElement('button');
    masterBtn.style.cssText = btnBase;
    masterBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="rgba(255,255,255,0.9)" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M18.36 6.64a9 9 0 1 1-12.73 0"></path><line x1="12" y1="2" x2="12" y2="12"></line></svg>`;
    
    masterBtn.onmouseenter = () => {
        if (isMasterEnabled) masterBtn.style.cssText = btnBase + btnHover;
    };
    masterBtn.onmouseleave = () => {
        if (isMasterEnabled) masterBtn.style.cssText = btnBase;
    };

    masterBtn.onclick = () => {
        isMasterEnabled = !isMasterEnabled;
        
        const nukeStyle = document.getElementById('ig-nuke-ui');

        if (!isMasterEnabled) {
            // TURN OFF SCRIPT (Normal Mode ON)
            globalStyle.disabled = true;
            ratioStyle.disabled = true;
            if (nukeStyle) nukeStyle.disabled = true;

            // Hide other UI controls
            saveBtn.style.display = 'none';
            hideBtn.style.display = 'none';
            ratioBtn.style.display = 'none';
            speedBtn.style.display = 'none';

            // Revert inline styles hidden by background engine
            document.querySelectorAll('[data-ig-script-hidden="true"]').forEach(el => {
                el.style.removeProperty('display');
                el.style.removeProperty('height');
                el.style.removeProperty('pointer-events');
                el.removeAttribute('data-ig-script-hidden');
            });

            // Reset video speeds to normal
            document.querySelectorAll('video').forEach(video => {
                video.playbackRate = 1;
            });

            masterBtn.style.opacity = '1';
            masterBtn.style.color = '#ff4444'; // Visual cue for disabled script
            masterBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="#ff4444" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M18.36 6.64a9 9 0 1 1-12.73 0"></path><line x1="12" y1="2" x2="12" y2="12"></line></svg>`;
            
            showToast('Normal Mode: ON');
        } else {
            // TURN ON SCRIPT (Normal Mode OFF)
            globalStyle.disabled = false;
            ratioStyle.disabled = false;
            if (nukeStyle) nukeStyle.disabled = !isNuked;

            // Show other UI controls
            saveBtn.style.display = 'flex';
            hideBtn.style.display = 'flex';
            ratioBtn.style.display = 'flex';
            speedBtn.style.display = 'flex';

            // Restore desired video speed
            document.querySelectorAll('video').forEach(video => {
                video.playbackRate = currentPlaybackSpeed;
            });

            masterBtn.style.opacity = '0.4';
            masterBtn.style.color = '';
            masterBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="rgba(255,255,255,0.9)" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M18.36 6.64a9 9 0 1 1-12.73 0"></path><line x1="12" y1="2" x2="12" y2="12"></line></svg>`;

            showToast('Normal Mode: OFF');
        }

        masterBtn.style.transform = 'scale(0.9)';
        setTimeout(() => masterBtn.style.transform = 'scale(1)', 150);
    };
    panel.appendChild(masterBtn);

    document.body.appendChild(panel);
})();

// ==UserScript==
// @name         Dictionary
// @namespace    https://viayoo.com/
// @version      0.1
// @description  try to take over the world!
// @author       You
// @run-at       document-end
// @match        https://*/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const popup = document.createElement('div');
    popup.style.cssText = `
        position: fixed; 
        left: 50%; 
        transform: translate(-50%, -50%); 
        background: #000000; 
        color: #ffffff; 
        padding: 28px; 
        border-radius: 16px; 
        border: 1px solid #333333;
        box-shadow: 0 10px 40px rgba(0,0,0,0.8); 
        z-index: 2147483647; 
        width: 90%; 
        max-width: 400px; 
        font-family: "Google Sans", Roboto, Arial, sans-serif; 
        display: none; 
        opacity: 0; 
        transition: opacity 0.2s ease; 
        text-align: left; 
        pointer-events: auto; 
        word-wrap: break-word;
        -webkit-user-select: none !important; 
        -moz-user-select: none !important;
        -ms-user-select: none !important;
        user-select: none !important;
        -webkit-touch-callout: none !important;
    `;
    document.body.appendChild(popup);

    let hideTimeout;
    let fetchTimeout;
    let lastWord = '';

    document.addEventListener('selectionchange', () => {
        const selection = window.getSelection();
        const text = selection.toString().trim();
        
        if (text.length > 1 && /^[A-Za-z]+$/.test(text)) {
            if (text !== lastWord) {
                lastWord = text;

                // --- DYNAMIC POSITIONING ---
                if (selection.rangeCount > 0) {
                    const range = selection.getRangeAt(0);
                    const rect = range.getBoundingClientRect();
                    const viewportHeight = window.innerHeight;
                    
                    // If text is in the lower half of the screen, show popup near the top.
                    // If text is in the upper half, show popup near the bottom.
                    if (rect.top > viewportHeight / 2) {
                        popup.style.top = '20%'; 
                    } else {
                        popup.style.top = '80%';
                    }
                }
                
                clearTimeout(fetchTimeout);
                fetchTimeout = setTimeout(() => {
                    fetchMeaning(text);
                }, 400); 
            }
        } else if (text.length === 0) {
            hidePopup();
            lastWord = '';
        }
    });

    popup.addEventListener('contextmenu', (e) => {
        e.preventDefault();
        e.stopPropagation();
    });

    popup.addEventListener('pointerdown', (e) => {
        e.preventDefault();
        e.stopPropagation();
        window.getSelection().removeAllRanges();
        hidePopup();
        lastWord = '';
    });

    async function fetchMeaning(word) {
        showPopup(`<div style="font-size:22px; font-weight:500; color:#9aa0a6;">Loading...</div>`);
        
        let hindiMeaning = '';
        try {
            const translateRes = await fetch(`https://translate.googleapis.com/translate_a/single?client=gtx&sl=en&tl=hi&dt=t&q=${word}`);
            if (translateRes.ok) {
                const translateData = await translateRes.json();
                hindiMeaning = translateData[0][0][0];
                if (hindiMeaning && hindiMeaning.toLowerCase() !== word.toLowerCase()) {
                    hindiMeaning = `<div style="font-size:22px; font-weight:500; color:#4caf50; margin-bottom:8px;">${hindiMeaning}</div>`;
                } else {
                    hindiMeaning = '';
                }
            }
        } catch (e) {
            // Ignore translation errors to ensure the English dictionary still works
        }

        let imageHtml = '';
        try {
            // Fetch relevant image from Wikipedia's summary API
            const wikiImgRes = await fetch(`https://en.wikipedia.org/api/rest_v1/page/summary/${word}`);
            if (wikiImgRes.ok) {
                const wikiImgData = await wikiImgRes.json();
                if (wikiImgData.thumbnail && wikiImgData.thumbnail.source) {
                    imageHtml = `<img src="${wikiImgData.thumbnail.source}" onerror="this.style.display='none'" style="max-width: 100%; max-height: 140px; border-radius: 8px; margin-top: 8px; margin-bottom: 12px; display: block; object-fit: cover;">`;
                }
            }
        } catch (e) {
            // Ignore image fetch errors
        }

        try {
            const response = await fetch(`https://api.dictionaryapi.dev/api/v2/entries/en/${word}`);
            if (response.ok) {
                const data = await response.json();
                const phonetic = data[0].phonetic ? `<span style="margin-left:10px; font-weight:normal; font-size:18px; color:#9aa0a6;">${data[0].phonetic}</span>` : '';
                const pos = data[0].meanings[0].partOfSpeech ? `<div style="font-size:18px; color:#8ab4f8; font-style:italic; margin-bottom:14px; text-transform:capitalize;">${data[0].meanings[0].partOfSpeech}</div>` : '';
                const definition = data[0].meanings[0].definitions[0].definition;
                
                showPopup(`
                    <div style="font-size:28px; font-weight:400; margin-bottom:6px; color:#ffffff;">${word}${phonetic}</div>
                    ${hindiMeaning}
                    ${imageHtml}
                    ${pos}
                    <div style="font-size:20px; line-height:1.5; color:#e8eaed;">${definition}</div>
                `);
                return;
            } else if (response.status === 404) {
                showNotFound(word, hindiMeaning, imageHtml);
                return;
            }
        } catch (error) {
            try {
                const wikiResponse = await fetch(`https://en.wiktionary.org/api/rest_v1/page/definition/${word}`);
                if (wikiResponse.ok) {
                    const wikiData = await wikiResponse.json();
                    const pos = wikiData.en[0].partOfSpeech ? `<div style="font-size:18px; color:#8ab4f8; font-style:italic; margin-bottom:14px; text-transform:capitalize;">${wikiData.en[0].partOfSpeech}</div>` : '';
                    let definitionHtml = wikiData.en[0].definitions[0].definition;
                    let cleanDef = definitionHtml.replace(/<[^>]*>?/gm, ''); 
                    
                    showPopup(`
                        <div style="font-size:28px; font-weight:400; margin-bottom:6px; color:#ffffff;">${word}</div>
                        ${hindiMeaning}
                        ${imageHtml}
                        ${pos}
                        <div style="font-size:20px; line-height:1.5; color:#e8eaed;">${cleanDef}</div>
                    `);
                    return;
                } else {
                    showNotFound(word, hindiMeaning, imageHtml);
                    return;
                }
            } catch (wikiError) {
                showPopup(`
                    <div style="color:#f28b82; font-size:22px; font-weight:500; margin-bottom:6px;">Connection Blocked</div>
                    <div style="font-size:18px; color:#e8eaed;">This website's strict security (CSP) blocks all external dictionary lookups.</div>
                `);
            }
        }
    }

    function showNotFound(word, hindiMeaning = '', imageHtml = '') {
        showPopup(`
            <div style="font-size:28px; font-weight:400; margin-bottom:10px; color:#ffffff;">${word}</div>
            ${hindiMeaning}
            ${imageHtml}
            <div style="font-size:20px; color:#9aa0a6;">Definition not found.</div>
        `);
    }

    function showPopup(htmlContent) {
        popup.innerHTML = htmlContent;
        popup.style.display = 'block';
        setTimeout(() => popup.style.opacity = '1', 10);
        
        clearTimeout(hideTimeout);
        hideTimeout = setTimeout(hidePopup, 8000); 
    }

    function hidePopup() {
        popup.style.opacity = '0';
        setTimeout(() => {
            popup.style.display = 'none';
        }, 200);
    }
})();
